Your Task Implement our beloved Array,prototype.filter from scratch. 

            
              Array.prototype.filter = function(callback, args) {
                const len = this.length;
                const newArr = new Array();
                for(let i = 0; i < len; i++) {
                  let res = callback.call(args, this[i], i, this);
                  if(i in this) {
                    if(res) {
                      newArr.push(this[i]);
                    }
                  }
                }
                return newArr;
              }
